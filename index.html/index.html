.leaflet-popup-content { font-family: 'Georgia', serif; font-size: 14px; color: #333; } // Basemaps var satellite = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution: '© Esri' }); var topo = L.tileLayer('https://{s}.tile.opentopomap.org/{z}/{x}/{y}.png', { attribution: '© OpenTopoMap' }); var streets = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap' }); // Map init var map = L.map('map', { center: [28.555, -82.387], zoom: 7, layers: [streets] }); // Layer control var baseMaps = { "Satellite": satellite, "Topographic": topo, "Streets": streets }; L.control.layers(baseMaps).addTo(map); // Ley line overlay placeholder fetch('data/ley_lines.geojson') .then(res => res.json()) .then(data => { L.geoJSON(data, { style: { color: '#8B00FF', weight: 2, opacity: 0.8, dashArray: '5, 5' }, onEachFeature: function (feature, layer) { if (feature.properties && feature.properties.name) { layer.bindPopup(`Ley Line: ${feature.properties.name}`); } } }).addTo(map); });
